Q1:-"Problem: You are given a string (STR) of length N, consisting of only the lower case English alphabet. Your task is to remove all the duplicate occurrences of characters in the string. Input: abcadeecfb Output: abcdef"

Function:-
        function removeDuplicates(str) {
          const uniqueChars = {};
        
          // Use map() to loop through the string
          str.split('').map(char => {
            uniqueChars[char] = true;
          });
        
          // Use Object.keys() to get an array of the unique characters
          const uniqueArray = Object.keys(uniqueChars);
        
          // Join the array back into a string and return it
          return uniqueArray.join('');
        }
        
        
        const str = 'abcadeecfb';
        const result = removeDuplicates(str);
        console.log(result);

Q2:-"Problem: You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps) Input: abcadeecfb Output: a=2 b=2 c=2 d=1 e=2 f=1"

Function:-var string = "abcadeecfb";
        const myMap = new Map();
        for (var alphabet of string) {
          if (myMap.has(alphabet)) {
            myMap.set(alphabet, myMap.get(alphabet) + 1);
          } else {
            myMap.set(alphabet, 1);
          }
        }
